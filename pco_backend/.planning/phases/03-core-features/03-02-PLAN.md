    app/services/interest_service.py:
    - submit_interest(db: Session, payload: InterestFormCreate) -> InterestSubmission:
      existing = db.query(InterestSubmission).filter(InterestSubmission.email == payload.email).first()
      if existing: raise HTTPException(409, "Email already submitted")
      obj = InterestSubmission(**payload.model_dump()); db.add(obj); db.commit(); db.refresh(obj); return obj
    - list_submissions(db: Session) -> list[InterestSubmission]: return db.query(InterestSubmission).all()

    app/services/rush_service.py:
    - get_rush(db: Session) -> RushInfo | None: return db.query(RushInfo).first()
    - upsert_rush(db: Session, payload: RushInfoUpdate) -> RushInfo:
      rush = db.query(RushInfo).first()
      if rush is None: rush = RushInfo(); db.add(rush)
      rush.dates = payload.dates; rush.times = payload.times; rush.locations = payload.locations; rush.description = payload.description
      db.commit(); db.refresh(rush); return rush
    - toggle_visibility(db: Session) -> RushInfo:
      rush = db.query(RushInfo).first()
      if rush is None: rush = RushInfo(); db.add(rush)
      rush.is_published = not rush.is_published
      db.commit(); db.refresh(rush); return rush
