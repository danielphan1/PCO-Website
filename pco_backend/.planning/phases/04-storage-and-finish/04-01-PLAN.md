  <behavior>

    - delete_event attempts storage removal with try/except, logs warning on failure (import logging), then always deletes DB record and returns None

  <action>
    Create app/services/event_service.py:

    Imports: uuid, logging, date from datetime, Session from sqlalchemy.orm,
    HTTPException + status from fastapi, EventPDF from app.models.event_pdf,
    storage_service from app.storage.files, event_pdf_path from app.storage.paths

    Constants: MAX_SIZE_BYTES = 10 * 1024 * 1024
    logger = logging.getLogger(__name__)

    def list_events(db: Session) -> list[dict]:
      Query EventPDF ordered by EventPDF.date.desc().all()
      For each event, call storage_service.create_signed_url(event.storage_path, expires_in=3600)
      Return list of dicts: {"id": str(event.id), "title": event.title, "date": str(event.date), "url": url}
      url is None if create_signed_url returned None — do not raise

    def upload_event(db: Session, title: str, event_date: date, data: bytes, uploader_id: uuid.UUID) -> tuple[EventPDF, str | None]:
      1. Validate size: if len(data) > MAX_SIZE_BYTES: raise HTTPException(413, "File exceeds 10 MB limit.")
      2. Validate magic bytes: if data[:4] != b"%PDF": raise HTTPException(422, "File is not a valid PDF (missing %PDF header).")
      3. Generate event_id = uuid.uuid4(); path = event_pdf_path(event_id)
      4. storage_service.upload(path, data)  # storage-first (CONTEXT.md decision)
      5. try: create EventPDF(id=event_id, title=title, date=event_date, storage_path=path, uploaded_by=uploader_id)
             db.add(event); db.commit(); db.refresh(event)
         except Exception:
             try: storage_service.remove(path)  # best-effort rollback
             except Exception: pass
             raise HTTPException(500, "Failed to save event record.")
      6. Generate signed URL: url = storage_service.create_signed_url(path, expires_in=3600)
         (Returns None if URL generation fails — do not raise)
      7. return event, url

      NOTE: Generating the signed URL inside the service keeps all storage_service calls in one
      namespace (app.services.event_service.storage_service). The router patches this single
      namespace and never needs to import or call storage_service directly.

    def delete_event(db: Session, event_id: uuid.UUID) -> None:
      event = db.query(EventPDF).filter(EventPDF.id == event_id).first()
      if event is None: raise HTTPException(404, "Event not found.")
      try: storage_service.remove(event.storage_path)
      except Exception as exc: logger.warning("Storage removal failed for %s: %s", event.storage_path, exc)
      db.delete(event); db.commit()
  </action>
